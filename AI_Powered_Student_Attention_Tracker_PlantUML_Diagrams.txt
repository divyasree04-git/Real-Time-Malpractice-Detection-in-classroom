@startuml Class_Diagram
!theme blueprint
title AI-Powered Student Attention Tracker - Class Diagram

class AttentionTracker {
    -camera: VideoCapture
    -face_detector: FaceDetector
    -eye_tracker: EyeTracker
    -pose_estimator: PoseEstimator
    -ml_engine: MLEngine
    -database: Database
    +start_tracking(): void
    +stop_tracking(): void
    +get_attention_score(): float
    +configure_alerts(): void
}

class FaceDetector {
    -cascade_classifier: CascadeClassifier
    -detection_threshold: float
    +detect_faces(frame): List[Rectangle]
    +extract_face_region(frame, face_rect): Mat
    +calculate_face_score(face_region): float
}

class EyeTracker {
    -eye_cascade: CascadeClassifier
    -blink_threshold: float
    -ear_threshold: float
    +detect_eyes(face_region): List[Point]
    +calculate_ear(eye_landmarks): float
    +detect_blink(): boolean
    +get_eye_score(): float
}

class PoseEstimator {
    -model_points: List[Point3D]
    -camera_matrix: Mat
    -dist_coeffs: Mat
    +estimate_pose(landmarks): Tuple[float, float, float]
    +calculate_pose_score(yaw, pitch, roll): float
}

class MLEngine {
    -weights: Dict[string, float]
    -attention_model: Model
    +calculate_attention_score(face_score, eye_score, pose_score): float
    +update_weights(new_weights): void
    +predict_attention_level(features): float
}

class AlertSystem {
    -thresholds: Dict[string, float]
    -audio_enabled: boolean
    -visual_enabled: boolean
    +check_attention_level(score): void
    +trigger_alert(alert_type): void
    +configure_thresholds(thresholds): void
}

class Database {
    -connection: SQLiteConnection
    -session_table: string
    -analytics_table: string
    +insert_session_data(session_data): void
    +get_session_history(user_id): List[SessionData]
    +generate_analytics(): AnalyticsData
}

class AnalyticsEngine {
    -database: Database
    -report_generator: ReportGenerator
    +calculate_trends(): TrendData
    +generate_session_report(session_id): Report
    +export_data(format): string
}

class SessionData {
    +session_id: string
    +user_id: string
    +start_time: DateTime
    +end_time: DateTime
    +attention_scores: List[float]
    +alerts_triggered: int
}

AttentionTracker --> FaceDetector
AttentionTracker --> EyeTracker
AttentionTracker --> PoseEstimator
AttentionTracker --> MLEngine
AttentionTracker --> AlertSystem
AttentionTracker --> Database
AttentionTracker --> AnalyticsEngine
Database --> SessionData
AnalyticsEngine --> Database

@enduml

@startuml Sequence_Diagram_Attention_Monitoring
!theme blueprint
title Attention Monitoring Process - Sequence Diagram

actor Student
participant "Attention Tracker" as AT
participant "Face Detector" as FD
participant "Eye Tracker" as ET
participant "Pose Estimator" as PE
participant "ML Engine" as ML
participant "Alert System" as AS
participant "Database" as DB

Student -> AT: Start monitoring session
activate AT

AT -> FD: Initialize face detection
activate FD
FD --> AT: Face detector ready
deactivate FD

AT -> ET: Initialize eye tracking
activate ET
ET --> AT: Eye tracker ready
deactivate ET

AT -> PE: Initialize pose estimation
activate PE
PE --> AT: Pose estimator ready
deactivate PE

loop Real-time monitoring
    AT -> FD: Process video frame
    activate FD
    FD --> AT: Face regions detected
    deactivate FD
    
    AT -> ET: Analyze eye regions
    activate ET
    ET --> AT: Eye tracking data
    deactivate ET
    
    AT -> PE: Estimate head pose
    activate PE
    PE --> AT: Pose angles
    deactivate PE
    
    AT -> ML: Calculate attention score
    activate ML
    ML --> AT: Attention score (0-100)
    deactivate ML
    
    AT -> AS: Check alert thresholds
    activate AS
    
    alt Attention below threshold
        AS -> Student: Trigger alert notification
        AS -> DB: Log alert event
    end
    deactivate AS
    
    AT -> DB: Store session data
    activate DB
    DB --> AT: Data stored
    deactivate DB
end

Student -> AT: End monitoring session
AT -> DB: Save final session report
deactivate AT

@enduml

@startuml Activity_Diagram
!theme blueprint
title Attention Monitoring Session - Activity Diagram

start

:Initialize System Components;

:Load User Configuration;

if (Camera Available?) then (yes)
    :Initialize Camera Feed;
else (no)
    :Display Error Message;
    stop
endif

:Start Monitoring Session;

repeat
    :Capture Video Frame;
    
    if (Face Detected?) then (yes)
        :Extract Face Region;
        
        fork
            :Detect Eyes;
            :Calculate Eye Score;
        fork again
            :Estimate Head Pose;
            :Calculate Pose Score;
        end fork
        
        :Calculate Overall Attention Score;
        
        if (Score Below Threshold?) then (yes)
            :Trigger Alert;
            :Log Alert Event;
        else (no)
        endif
        
        :Update Real-time Display;
        :Store Frame Data;
        
    else (no)
        :Log No Face Detected;
    endif
    
repeat while (Session Active?)

:Generate Session Summary;

:Save Session to Database;

:Display Final Report;

stop
