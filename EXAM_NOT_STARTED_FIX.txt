function startClass() {
    fetch('/start_class')
        .then(response => response.json())
        .then(data => {
            if (data.status === 'success') {
                document.getElementById('startBtn').disabled = true;
                document.getElementById('stopBtn').disabled = false;
                alert('Exam Started! Monitoring is now active.');
            } else {
                alert('Error: ' + data.message);
            }
        })
        .catch(error => {
            console.error('Error:', error);
            alert('Failed to start exam');
        });
}

function stopClass() {
    fetch('/stop_class')
        .then(response => response.json())
        .then(data => {                    ? FIXED!
            if (data.status === 'success') {
                document.getElementById('startBtn').disabled = false;
                document.getElementById('stopBtn').disabled = true;
                window.location.href = '/dashboard';
            } else {
                alert('Error: ' + data.message);
            }
        })
        .catch(error => {
            console.error('Error:', error);
            alert('Failed to stop exam');
        });
}

                    WHY THIS ISSUE EXISTED
???????????????????????????????????????????????????????????????????????????????

The JavaScript error was likely introduced during previous code edits where
code was hastily pasted or modified without proper syntax checking.

The syntax error:
  .then data =>  (missing parentheses)

Should have been:
  .then(data) => (correct arrow function)

This prevented the Promise chain from executing properly, which meant:
1. The fetch response wasn't processed
2. Button states weren't updated
3. Frontend didn't know exam had started
4. Video feed continued showing "Exam Not Started" frame
