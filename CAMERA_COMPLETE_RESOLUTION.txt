def init_camera():
    """Initialize camera with error handling"""
    global cap
    try:
        cap = cv2.VideoCapture(0)
        if not cap.isOpened():
            cap = None
            return False
        return True
    except:
        cap = None
        return False

def get_camera():
    """Get camera instance, initialize if needed"""
    global cap
    if cap is None:
        init_camera()
    return cap
